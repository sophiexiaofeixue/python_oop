class Circle:
    VALID_COLORS = ('red', 'green', 'blue')
    def __init__(self, radius, color):
        self._radius = radius
        self._color = color
    
    def get_radius(self):
        return self._radius
    
    def set_radius(self, new_radius):
        if isinstance(new_radius, int) and new_radius > 0:
            self._radius = new_radius
        else:
            print('please enter a valida radius')
    
    radius = property(get_radius, set_radius)
    def get_color(self):
        return self._color
    def set_color(self, new_color):
        if new_color in Circle.VALID_COLORS:
            self._color = new_color
        else:
            print('pick a valid color')
    color = property(get_color, set_color)

my_circle = Circle(10, 'blue')
print(my_circle.radius)
print(my_circle.get_radius())
my_circle.radius = 16
print(my_circle.get_radius())
my_circle.color = 'red'
print(my_circle.color)
my_circle.color = 'white'
print(my_circle.color)
